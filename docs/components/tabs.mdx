# Tabs

A tabs component that replicates the Windows 95 tab interface with raised tab buttons.

## Installation

<InstallationTabs 
  componentName="tabs" 
  dependencies={["@radix-ui/react-tabs"]}
/>

## Usage

<AutoPreview>
  <Win95Tabs defaultValue="tab1" className="w-80">
    <Win95TabsList>
      <Win95TabsTrigger value="tab1">General</Win95TabsTrigger>
      <Win95TabsTrigger value="tab2">Advanced</Win95TabsTrigger>
      <Win95TabsTrigger value="tab3">Security</Win95TabsTrigger>
    </Win95TabsList>
    <Win95TabsContent value="tab1" className="p-4">
      <p className="text-xs text-win95-black mb-2">General settings and options.</p>
      <Win95Button size="sm">Apply</Win95Button>
    </Win95TabsContent>
    <Win95TabsContent value="tab2" className="p-4">
      <p className="text-xs text-win95-black mb-2">Advanced configuration options.</p>
      <Win95Button size="sm">Configure</Win95Button>
    </Win95TabsContent>
    <Win95TabsContent value="tab3" className="p-4">
      <p className="text-xs text-win95-black mb-2">Security and privacy settings.</p>
      <Win95Button size="sm">Update</Win95Button>
    </Win95TabsContent>
  </Win95Tabs>
</AutoPreview>
