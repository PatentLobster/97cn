```tsx

  Win95Tabs, 
  Win95TabsList, 
  Win95TabsTrigger, 
  Win95TabsContent 
} from "@/components/97cn/tabs"

export function Example() {
  return (
    <Win95Tabs defaultValue="tab1">
      <Win95TabsList>
        <Win95TabsTrigger value="tab1">General</Win95TabsTrigger>
        <Win95TabsTrigger value="tab2">Advanced</Win95TabsTrigger>

      <Win95TabsContent value="tab1">
        General settings content
      </Win95TabsContent>
      <Win95TabsContent value="tab2">
        Advanced settings content
      </Win95TabsContent>
    </Win95Tabs>
  )
}
```

## Props

### Win95Tabs

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| value | string | - | Controlled active tab |
| defaultValue | string | - | Default active tab |
| onValueChange | (value: string) => void | - | Callback when tab changes |
| orientation | "horizontal" \| "vertical" | "horizontal" | Tab orientation |

### Win95TabsTrigger

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| value | string | - | Tab value |
| disabled | boolean | false | Whether the tab is disabled |

### Win95TabsContent

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| value | string | - | Associated tab value |
